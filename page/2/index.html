class BigInt {
public:
    // Тъёжђа
    BigInt() : BigInt(0) {}
    BigInt(int x) {
        if (x < 0) negative = true, x = -x;
        else negative = false;
        if (x == 0) digits.push_back(0);
        else while (x) { digits.push_back(x % 10); x /= 10; }
    }
    BigInt(const std::string &s) {
        int start = 0;
        if (s[0] == '-') { negative = true; start = 1; }
        else negative = false;
        for (int i = s.size() - 1; i >= start; --i) {
            if (!isdigit(s[i])) throw std::invalid_argument("not a number");
            digits.push_back(s[i] - '0');
        }
        remove_leading_zeros();
        if (digits.empty()) digits.push_back(0), negative = false;
    }
    BigInt(const char *s) : BigInt(std::string(s)) {}

    // Т»ћУЙЃУ┐љу«Ќугд№╝ѕу╗Ют»╣тђ╝тЁѕТ»ћУЙЃжЋ┐т║дтєЇжђљСйЇ№╝Ѕ
    bool operator==(const BigInt &rhs) const {
        return negative == rhs.negative && digits == rhs.digits;
    }
    bool operator!=(const BigInt &rhs) const { return !(*this == rhs); }
    bool operator<(const BigInt &rhs) const {
        if (negative != rhs.negative) return negative;
        if (digits.size() != rhs.digits.size())
            return (digits.size() < rhs.digits.size()) ^ negative;
        for (int i = digits.size() - 1; i >= 0; --i)
            if (digits[i] != rhs.digits[i])
                return (digits[i] < rhs.digits[i]) ^ negative;
        return false;
    }
    bool operator>(const BigInt &rhs) const { return rhs < *this; }
    bool operator<=(const BigInt &rhs) const { return !(rhs < *this); }
    bool operator>=(const BigInt &rhs) const { return !(*this < rhs); }

    // у«ЌТю»У┐љу«Ќугд
    BigInt operator-() const {
        BigInt res = *this;
        if (!is_zero()) res.negative = !negative;
        return res;
    }
    BigInt operator+(const BigInt &rhs) const {
        if (negative == rhs.negative) {
            BigInt res = add_abs(*this, rhs);
            res.negative = negative;
            return res;
        } else {
            if (abs_compare(*this, rhs) >= 0) {
                BigInt res = sub_abs(*this, rhs);
                res.negative = negative;
                return res;
            } else {
                BigInt res = sub_abs(rhs, *this);
                res.negative = rhs.negative;
                return res;
            }
        }
    }
    BigInt operator-(const BigInt &rhs) const {
        return *this + (-rhs);
    }
    BigInt operator*(const BigInt &rhs) const {
        BigInt res;
        res.digits.assign(digits.size() + rhs.digits.size(), 0);
        for (size_t i = 0; i < digits.size(); ++i) {
            int carry = 0;
            for (size_t j = 0; j < rhs.digits.size(); ++j) {
                carry += res.digits[i + j] + digits[i] * rhs.digits[j];
                res.digits[i + j] = carry % 10;
                carry /= 10;
            }
            if (carry) res.digits[i + rhs.digits.size()] += carry;
        }
        res.remove_leading_zeros();
        res.negative = negative != rhs.negative;
        if (res.is_zero()) res.negative = false;
        return res;
    }
    BigInt operator/(const BigInt &rhs) const {
        if (rhs.is_zero()) throw std::domain_error("division by zero");
        if (abs_compare(*this, rhs) < 0) return BigInt(0);
        BigInt res, remainder;
        divide_abs(*this, rhs, res, remainder);
        res.negative = negative != rhs.negative;
        if (res.is_zero()) res.negative = false;
        return res;
    }
    BigInt operator%(const BigInt &rhs) const {
        if (rhs.is_zero()) throw std::domain_error("modulo by zero");
        BigInt res, remainder;
        divide_abs(*this, rhs, res, remainder);
        remainder.negative = negative;
        if (remainder.is_zero()) remainder.negative = false;
        return remainder;
    }

    // тцЇтљѕУхІтђ╝
    BigInt& operator+=(const BigInt &rhs) { *this = *this + rhs; return *this; }
    BigInt& operator-=(const BigInt &rhs) { *this = *this - rhs; return *this; }
    BigInt& operator*=(const BigInt &rhs) { *this = *this * rhs; return *this; }
    BigInt& operator/=(const BigInt &rhs) { *this = *this / rhs; return *this; }
    BigInt& operator%=(const BigInt &rhs) { *this = *this % rhs; return *this; }

    // УЙЊтЄ║ТхЂ
    friend std::ostream& operator<<(std::ostream &os, const BigInt &a) {
        if (a.negative && !a.is_zero()) os << '-';
        for (int i = (int)a.digits.size() - 1; i >= 0; --i)
            os << a.digits[i];
        return os;
    }

    // УйгТЇбСИ║тГЌугдСИ▓
    std::string to_string() const {
        std::string res;
        if (negative && !is_zero()) res += '-';
        for (int i = (int)digits.size() - 1; i >= 0; --i)
            res += char('0' + digits[i]);
        return res;
    }

private:
    std::vector<int> digits;  // СйјСйЇтюетЅЇ№╝їТ»ЈСИфтЁЃу┤а0-9
    bool negative = false;

    // тј╗ТјЅтЅЇт»╝жЏХ
    void remove_leading_zeros() {
        while (digits.size() > 1 && digits.back() == 0)
            digits.pop_back();
        if (digits.empty()) digits.push_back(0);
    }

    bool is_zero() const {
        return digits.size() == 1 && digits[0] == 0;
    }

    // у╗Ют»╣тђ╝тіаТ│Ћ
    static BigInt add_abs(const BigInt &a, const BigInt &b) {
        BigInt res;
        res.digits.clear();
        int carry = 0;
        size_t i = 0;
        while (i < a.digits.size() || i < b.digits.size() || carry) {
            int sum = carry;
            if (i < a.digits.size()) sum += a.digits[i];
            if (i < b.digits.size()) sum += b.digits[i];
            res.digits.push_back(sum % 10);
            carry = sum / 10;
            ++i;
        }
        return res;
    }

    // у╗Ют»╣тђ╝тЄЈТ│Ћ№╝ѕУдЂТ▒ѓ a >= b СИћ a,b жЮъУ┤Ъ№╝Ѕ
    static BigInt sub_abs(const BigInt &a, const BigInt &b) {
        BigInt res;
        res.digits.clear();
        int borrow = 0;
        for (size_t i = 0; i < a.digits.size(); ++i) {
            int diff = a.digits[i] - borrow;
            if (i < b.digits.size()) diff -= b.digits[i];
            if (diff < 0) { diff += 10; borrow = 1; }
            else borrow = 0;
            res.digits.push_back(diff);
        }
        res.remove_leading_zeros();
        return res;
    }

    // у╗Ют»╣тђ╝Т»ћУЙЃ№╝џУ┐ћтЏъ 1 УІЦ |a| > |b|, -1 УІЦ <, 0 УІЦуЏИуГЅ
    static int abs_compare(const BigInt &a, const BigInt &b) {
        if (a.digits.size() != b.digits.size())
            return a.digits.size() < b.digits.size() ? -1 : 1;
        for (int i = (int)a.digits.size() - 1; i >= 0; --i)
            if (a.digits[i] != b.digits[i])
                return a.digits[i] < b.digits[i] ? -1 : 1;
        return 0;
    }

    // жЎцТ│Ћ№╝ѕу╗Ют»╣тђ╝№╝Ѕ№╝їтЋєтњїСйЎТЋ░
    static void divide_abs(const BigInt &a, const BigInt &b, BigInt &quot, BigInt &rem) {
        quot.digits.clear();
        rem = BigInt(0);
        for (int i = (int)a.digits.size() - 1; i >= 0; --i) {
            rem.digits.insert(rem.digits.begin(), a.digits[i]);
            rem.remove_leading_zeros();
            // У»ЋтЋє
            int q = 0;
            int l = 0, r = 9;
            while (l <= r) {
                int mid = (l + r) / 2;
                BigInt t = b * BigInt(mid);
                if (abs_compare(rem, t) >= 0) {
                    q = mid;
                    l = mid + 1;
                } else {
                    r = mid - 1;
                }
            }
            quot.digits.insert(quot.digits.begin(), q);
            BigInt sub = b * BigInt(q);
            rem = sub_abs(rem, sub);
        }
        quot.remove_leading_zeros();
        rem.remove_leading_zeros();
    }
};
