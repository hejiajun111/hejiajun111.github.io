矩阵快速幂

2026/5/25 算法模板 c++ 算法模板 
#include <vector>
using Matrix = std::vector<std::vector<long long>>;
Matrix mat_mul(const Matrix& A, const Matrix& B, long long mod) {
    int n = A.size();
    Matrix C(n, std::vector<long long>(n, 0));
    for (int i = 0; i < n; ++i)
        for (int k = 0; k < n; ++k)
            if (A[i][k] != 0)
                for (int j = 0; j < n; ++j)
                    C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % mod;
    return C;
}
Matrix mat_pow(Matrix A, long long power, long long mod) {
    int n = A.size();
    Matrix res(n, std::vector<long long>(n, 0));
    for (int i = 0; i < n; ++i) res[i][i] = 1;  // 单位矩阵
    
    while (power) {
        if (power & 1) res = mat_mul(res, A, mod);
        A = mat_mul(A, A, mod);
        power >>= 1;
    }
    return res;
}
