珂朵莉树

2026/5/25 算法模板 c++ 算法模板 
struct Node {
    int l, r;
    mutable int v;          // mutable 允许在 set 中修改 v
    Node(int l, int r = -1, int v = 0) : l(l), r(r), v(v) {}
    bool operator<(const Node &o) const { return l < o.l; }
};

set<Node> odt;

// 将包含位置 pos 的区间 [l, r] 分裂为 [l, pos-1] 和 [pos, r]
auto split(int pos) {
    auto it = odt.lower_bound(Node(pos));
    if (it != odt.end() && it->l == pos) return it;
    --it;
    int l = it->l, r = it->r, v = it->v;
    odt.erase(it);
    odt.insert(Node(l, pos - 1, v));
    return odt.insert(Node(pos, r, v)).first;
}

// 区间赋值（推平）
void assign(int l, int r, int v) {
    auto itr = split(r + 1), itl = split(l);
    odt.erase(itl, itr);
    odt.insert(Node(l, r, v));
}

// 区间加
void add(int l, int r, int v) {
    auto itr = split(r + 1), itl = split(l);
    for (auto it = itl; it != itr; ++it) it->v += v;
}

// 区间第 k 小
int kth(int l, int r, int k) {
    auto itr = split(r + 1), itl = split(l);
    vector<pair<int, int>> tmp;  // (值, 长度)
    for (auto it = itl; it != itr; ++it)
        tmp.emplace_back(it->v, it->r - it->l + 1);
    sort(tmp.begin(), tmp.end());
    for (auto [val, len] : tmp) {
        if (k <= len) return val;
        k -= len;
    }
    return -1;
}

// 快速幂（取模）
int qpow(int a, int b, int mod) {
    int res = 1;
    a %= mod;
    while (b) {
        if (b & 1) res = 1LL * res * a % mod;
        a = 1LL * a * a % mod;
        b >>= 1;
    }
    return res;
}

// 区间幂次和 ∑ (v^x) * len  对 mod 取模
int range_pow_sum(int l, int r, int x, int mod) {
    auto itr = split(r + 1), itl = split(l);
    int ans = 0;
    for (auto it = itl; it != itr; ++it) {
        int len = it->r - it->l + 1;
        ans = (ans + 1LL * len * qpow(it->v, x, mod)) % mod;
    }
    return ans;
}
