线段树

2026/5/25 算法模板 c++ 算法模板 
template<typename T>
class SegTree {
public:
    explicit SegTree(int n) : n(n), tree(4 * n), lazy(4 * n) {}
    // 建树（可选，初始全0时可省略）
    void build(const std::vector<T>& arr) {
        std::function<void(int, int, int)> dfs = [&](int u, int l, int r) {
            if (l == r) { tree[u] = arr[l - 1]; return; }
            int m = (l + r) / 2;
            dfs(u * 2, l, m);
            dfs(u * 2 + 1, m + 1, r);
            pull(u);
        };
        dfs(1, 1, n);
    }
    // 区间 [ql, qr] 增加 val
    void add(int ql, int qr, T val) {
        add(1, 1, n, ql, qr, val);
    }
    // 区间求和 [ql, qr]
    T sum(int ql, int qr) {
        return query(1, 1, n, ql, qr);
    }
    
private:
    int n;
    std::vector<T> tree, lazy;
    void pull(int u) { tree[u] = tree[u * 2] + tree[u * 2 + 1]; }
    void push(int u, int l, int r) {
        if (lazy[u] != 0 && l != r) {
            int m = (l + r) / 2;
            // 下传标记到左右儿子
            apply(u * 2, l, m, lazy[u]);
            apply(u * 2 + 1, m + 1, r, lazy[u]);
            lazy[u] = 0;
        }
    }
    void apply(int u, int l, int r, T val) {
        lazy[u] += val;
        tree[u] += val * (r - l + 1);
    }
    void add(int u, int l, int r, int ql, int qr, T val) {
        if (ql <= l && r <= qr) {
            apply(u, l, r, val);
            return;
        }
        push(u, l, r);
        int m = (l + r) / 2;
        if (ql <= m) add(u * 2, l, m, ql, qr, val);
        if (qr > m)  add(u * 2 + 1, m + 1, r, ql, qr, val);
        pull(u);
    }
    T query(int u, int l, int r, int ql, int qr) {
        if (ql <= l && r <= qr) return tree[u];
        push(u, l, r);
        int m = (l + r) / 2;
        T res = 0;
        if (ql <= m) res += query(u * 2, l, m, ql, qr);
        if (qr > m)  res += query(u * 2 + 1, m + 1, r, ql, qr);
        return res;
    }
};
