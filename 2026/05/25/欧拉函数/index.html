欧拉函数

2026/5/25 算法模板 c++ 算法模板 
struct EulerSieve {
    vector<int> primes;           // 素数列表
    vector<bool> is_composite;    // 是否为合数
    vector<int> min_factor;       // 最小质因子（可选）
    vector<int> phi;              // 欧拉函数（可选）

    // 构造函数，筛出 [0, n] 内的素数
    EulerSieve(int n) : is_composite(n + 1, false), min_factor(n + 1, 0), phi(n + 1, 0) {
        phi[1] = 1;
        for (int i = 2; i <= n; ++i) {
            if (!is_composite[i]) {
                primes.push_back(i);
                min_factor[i] = i;
                phi[i] = i - 1;          // 素数的欧拉函数 = i-1
            }
            for (int p : primes) {
                if (i * p > n) break;
                is_composite[i * p] = true;
                min_factor[i * p] = p;
                if (i % p == 0) {
                    // p 是 i 的最小质因子
                    phi[i * p] = phi[i] * p;
                    break;
                } else {
                    phi[i * p] = phi[i] * (p - 1);
                }
            }
        }
    }
    // 判断 x 是否为素数（x 需在筛的范围内）
    bool is_prime(int x) const {
        return !is_composite[x];
    }
};
