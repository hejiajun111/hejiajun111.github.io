// 需要类型 T 支持加法与减法（可用于前缀和）
template<typename T>
class Fenwick {
public:
    explicit Fenwick(int n) : n(n), bit(n + 1) {}
    // 单点增加
    void add(int i, T delta) {
        for (; i <= n; i += i & -i)
            bit[i] += delta;
    }
    // 前缀和 [1..i]
    T sum(int i) const {
        T res = 0;
        for (; i > 0; i -= i & -i)
            res += bit[i];
        return res;
    }
    // 区间和 [l..r]
    T query(int l, int r) const {
        return sum(r) - sum(l - 1);
    }
private:
    int n;
    std::vector<T> bit;
};
