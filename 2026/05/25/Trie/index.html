//字符串Trie
class Trie {
public:
    Trie() : nodes(1) {}  // 初始化根节点（下标0）

    // 插入字符串
    void insert(const std::string& word) {
        int u = 0;
        for (char ch : word) {
            int c = ch - 'a';
            if (nodes[u].next[c] == 0) {
                nodes[u].next[c] = nodes.size();
                nodes.emplace_back();   // 新建节点
            }
            u = nodes[u].next[c];
        }
        nodes[u].cnt++;   // 单词末尾计数（可表示重复次数）
    }

    // 查询单词是否存在（至少一次）
    bool search(const std::string& word) const {
        int u = find(word);
        return u != -1 && nodes[u].cnt > 0;
    }

    // 查询是否存在以给定前缀开头的单词
    bool startsWith(const std::string& prefix) const {
        return find(prefix) != -1;
    }

private:
    struct Node {
        int next[26] = {};   // 孩子指针，0表示空
        int cnt = 0;         // 以该节点结尾的单词个数
    };
    std::vector<Node> nodes;

    // 返回字符串对应路径的节点下标，若路径不存在则返回-1
    int find(const std::string& s) const {
        int u = 0;
        for (char ch : s) {
            int c = ch - 'a';
            if (nodes[u].next[c] == 0) return -1;
            u = nodes[u].next[c];
        }
        return u;
    }
};

//01Trie
class Trie01 {
public:
    Trie01() : nodes(1) {}  // 根节点下标0
    // 插入整数 x（二进制位从高位到低位）
    void insert(int x) {
        int u = 0;
        for (int i = MAX_BIT; i >= 0; --i) {
            int b = (x >> i) & 1;
            if (nodes[u].next[b] == 0) {
                nodes[u].next[b] = nodes.size();
                nodes.emplace_back();
            }
            u = nodes[u].next[b];
        }
        nodes[u].cnt++;  // 记录末尾次数（可选）
    }
    // 查询与 x 异或能得到的最大值
    int max_xor(int x) const {
        int u = 0, res = 0;
        for (int i = MAX_BIT; i >= 0; --i) {
            int b = (x >> i) & 1;
            int want = 1 - b;  // 期望相反的位
            if (nodes[u].next[want] != 0) {
                res |= (1 << i);
                u = nodes[u].next[want];
            } else {
                u = nodes[u].next[b];
            }
        }
        return res;
    }
private:
    static constexpr int MAX_BIT = 30;  // 处理 0~30 位，共31位（int范围）
    struct Node {
        int next[2] = {0, 0};
        int cnt = 0;
    };
    std::vector<Node> nodes;
};
